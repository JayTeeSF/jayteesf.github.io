REPL Recipes

All examples in this file can be pasted into `./bin/hey-repl`.

In an interactive terminal, use Up arrow to repeat earlier top-level commands

and Down arrow to move forward through history. The prompt starts in insert

mode; press Esc for Vim normal mode (`h`/`l`, `0`, `^`, `$`, `w`, `b`, `x`, `D`,

`d$`, `C`, `r<char>`) and use `i`, `a`, `I`, or `A` to resume insertion.

Hello

says 'Hello from the REPL'
quit

Values and comparisons

let name = 'Hey person'
let answer = 40 + 2
says 'Hello, ' + name
says 'Count: ' + answer
says answer == 42
says answer > 10
quit

JSON/hash object

let doc = {'name':'Hey','visits':2,'ok':true}
get(doc, 'name')
doc = {...doc, visits: get(doc, 'visits') + 1}
to_json(doc)
quit

JSON arrays

let xs = [1, 2]
xs = [...xs, {name:'Hey'}]
get(xs, 2)
to_json(xs)
quit

Modules around JSON values

module Users
  fn name(user)
    return get(user, 'name')
  end

  fn bump_seen(user)
    return {...user, seen: get(user, 'seen') + 1}
  end
end

let user = {name:'Ada', seen:1}
Users.name(user)
user = Users.bump_seen(user)
get(user, 'seen')
quit

Control flow

let count = 5
if count > 3
  says 'big'
else
  says 'small'
end
quit

Echo actor

actor echo
  receive msg
  says msg
end

let p = spawn echo
send p, 'ping'
stop p
quit

Counter actor with state

actor counter(label)
  state count = 0
  receive msg

  if msg == 'inc'
    set count = count + 1
    says label + ': ' + count
  else
    if msg == 'stop'
      exit
    end
  end
end

let visits = spawn counter('visits')
send visits, 'inc'
send visits, 'inc'
send visits, 'stop'
join visits
quit

Message chain

actor sink
  receive msg
  says msg
end

actor relay(next)
  receive msg
  send next, msg
end

let sink_actor = spawn sink
let third = spawn relay(sink_actor)
let second = spawn relay(third)
let first = spawn relay(second)

send first, 'around the chain'

stop first
stop second
stop third
stop sink_actor
quit

Real Hey process ring

actor ring_node(id)
  state next = nil
  receive msg

  if next == nil
    set next = msg
  else
    if msg == -1
      send next, -1
      exit
    else
      if msg > 0
        says 'node ' + id + ' hop ' + msg
        send next, msg - 1
      else
        says 'node ' + id + ' done'
        send next, -1
        exit
      end
    end
  end
end

let n1 = spawn ring_node(1)
let n2 = spawn ring_node(2)
let n3 = spawn ring_node(3)

send n1, n2
send n2, n3
send n3, n1

send n1, 5

join n1
join n2
join n3
quit
