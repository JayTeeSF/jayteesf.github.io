Web stdlib (`stdlib:web`)

`stdlib:web` is Hey's official web surface. Use it for normal application code:

import 'stdlib:web'

It sits above the lower-level `stdlib:http` parser/serializer and `stdlib:net` listener descriptors. The design goal is a small Rails-like request/router/response layer that keeps request handling readable, pure where possible, and ready to compose with actors and Jobs.

Why dispatch works without opening a socket

`web.dispatch(routes, request)` takes ordinary Hey values and returns an ordinary response value. That lets controller and routing specs run quickly and deterministically with no port conflicts, no sleeps, and no network flakiness.

A real service still uses sockets at the boundary. Build the executable descriptor with `web.service(routes, options)` and consume it with `web.start(service)`; `web.serve(routes, options)` is the convenience composition of those two operations. Most app behavior should be tested below that boundary; a smaller number of integration specs can exercise `web.serve`/listener plumbing.

Start a service descriptor

`web.service` is not decorative metadata. `web.start` validates and executes it:

let service = web.service(routes, {
  host: '127.0.0.1',
  port: 3072,
  workers: 8,
})

web.start(service)

For the common case, use the equivalent convenience call:

web.serve(routes, {host: '127.0.0.1', port: 3072, workers: 8})

A tiny route

spec 'web health route'
  expect stdout "200\n{\"ok\":true,\"service\":\"recall_coach\"}\n"
end

import 'stdlib:web'

module HealthController
  fn show(request)
    return web.json({ok: true, service: 'recall_coach'})
  end
end

program
  let routes = web.routes([
    web.get('/health', HealthController.show)
  ])

  let response = web.dispatch(routes, {
    method: 'GET',
    path: '/health',
    headers: {},
    body: ''
  })

  says response.status
  says response.body
end

Expected output:

200
{"ok":true,"service":"recall_coach"}

Requests

A request is an immutable object:

{
  method: 'POST',
  path: '/sessions/session_abc/responses',
  params: {session_id: 'session_abc'},
  query: {},
  headers: {'content-type': 'application/json'},
  body: '{"card_id":"card_1"}',
  request_id: 'req_123'
}

Common helpers:

web.method(request)
web.path(request)
web.params(request)
web.param(request, 'session_id')
web.query(request)
web.query_param(request, 'page', '1')
web.headers(request)
web.header(request, 'content-type')
web.body(request)
web.json_body(request)
web.user(request)

Responses

A response is an immutable object:

{status: 200, headers: {'content-type': 'application/json'}, body: '{"ok":true}'}

Use helpers instead of hand-building responses:

web.text('hello')
web.html('<h1>Hello</h1>')
web.json({ok: true})
web.created({id: 'card_1'})
web.accepted({queued: true})
web.bad_request({error: 'bad_request'})
web.unauthorized({error: 'unauthorized'})
web.forbidden({error: 'forbidden'})
web.not_found({error: 'not_found'})
web.conflict({error: 'conflict'})
web.unprocessable({error: 'invalid'})
web.server_error({error: 'server_error'})
web.redirect('/next')

Render a response as HTTP text when you need to inspect or hand it to lower-level HTTP tooling:

let response_text = web.response_text(web.text('ok'))

Route params and JSON bodies

import 'stdlib:web'

module SessionsController
  fn show(request)
    return web.text(web.param(request, 'session_id'))
  end
end

module ResponsesController
  fn create(request)
    let input = web.json_body(request)
    return web.text(input.card_id)
  end
end

program
  let routes = web.routes([
    web.get('/sessions/:session_id', SessionsController.show),
    web.post('/responses', ResponsesController.create)
  ])
end

A route miss returns `404`. A path match with the wrong method returns `405`.

Files from routes

For tutorials and small apps, route to files through `web.file(root, request_path)`. The helper delegates path safety to `stdlib:static_files`, rejecting traversal/dotfile/encoded unsafe paths before reading.

import 'stdlib:web'

module PublicFiles
  fn show(request)
    return web.file('public', web.param(request, 'name'))
  end
end

program
  let routes = web.routes([
    web.get('/files/:name', PublicFiles.show)
  ])
end

Middleware

module AppMiddleware
  fn add_app_header(request)
    return {...request, headers: {...request.headers, 'x-hey-app': 'recall'}}
  end
end

let app = web.stack([
  web.request_id(),
  web.recover_errors(),
  web.secure_headers(),
  web.max_body(1048576),
  web.middleware('app_header', AppMiddleware.add_app_header)
], routes)

The stack is the value you dispatch or serve; it is not decorative configuration. This keeps the same middleware behavior in unit tests and on the live socket boundary:

import 'stdlib:web'

module HealthController
  fn show(request)
    return web.json({ok: true, request_id: request.request_id})
  end
end

program
  let routes = web.routes([
    web.get('/health', HealthController.show)
  ])
  let app = web.stack([
    web.request_id(),
    web.recover_errors(),
    web.secure_headers(),
    web.max_body(1048576)
  ], routes)

  # Fast controller test without opening a socket.
  let response = web.dispatch(app, {
    method: 'GET', path: '/health', headers: {}, body: '', params: {}
  })
  says response.status
  says response.headers['x-content-type-options']

  # Real HTTP endpoint: serve the exact same stack over TCP.
  web.serve(app, {host: '127.0.0.1', port: 3072, workers: 4})
end

`web.middleware(name, fn)` is how applications construct middleware outside the

stdlib. A custom middleware function receives the immutable request and returns

the request passed to the next layer. Use spread updates to add application

headers, authentication context, tracing data, or other request-local values.

Current stdlib middleware behavior:

  • `web.request_id()` adds `request_id: 'req_1'` when a request is missing one.
  • `web.recover_errors()` converts handler failures into a `500` response.
  • `web.secure_headers()` adds conservative headers: `nosniff`, `DENY`, `no-referrer`, and `default-src 'self'`.
  • `web.max_body(bytes)` rejects oversized request bodies with `413`.
  • `web.require_json()` rejects bad JSON bodies with `400` before dispatch.
  • `web.log_requests()`, `web.rate_limit(policy)`, and `web.cors(policy)` are stable descriptors for upcoming runtime behavior.

Service boundary

Use port `3072` in examples and tutorials:

import 'stdlib:web'

program
  let routes = web.routes([])

  web.serve(routes, {
    host: '127.0.0.1',
    port: 3072,
    workers: 4,
    request_timeout_ms: 15000,
    max_body_bytes: 1048576
  })
end

`web.service(routes, options)` returns a composable descriptor. `web.serve(routes, options)` is now the live service boundary: it binds a TCP listener, accepts HTTP requests, dispatches them through the same route/controller path as `web.dispatch`, and writes HTTP/1.1 responses.

Jobs integration

Keep web requests small. Send expensive work to `stdlib:jobs` and return `202 Accepted`:

import 'stdlib:web'
import 'stdlib:jobs'

module EvaluateSessionWorker
  fn enqueue(input)
    return {id: 'job_' + input.id}
  end
end

module SessionsController
  fn finish(request)
    let session_id = web.param(request, 'session_id')
    let receipt = EvaluateSessionWorker.enqueue({id: session_id})

    return web.accepted({
      ok: true,
      session_id: session_id,
      evaluation_status: 'queued',
      receipt_id: receipt.id
    })
  end
end

Under the hood: routes, handler references, and dispatch

This route table stores handler references as values:

let routes = web.routes([
  web.get('/health', HealthController.show)
])

`HealthController.show` is not called when the route table is built. The compiler stores the module-qualified function name as the handler value. `web.dispatch(routes, request)` then:

1. scans the route descriptors in order,

2. matches method and path,

3. extracts `:params` into the request value,

4. calls the handler with that enriched request,

5. returns the handler's immutable response value.

That is why controller specs can call `web.dispatch(...)` without sockets. It is also why handlers should stay small: parse the request, validate it, enqueue expensive work into Jobs, and return a response.

Long-running services and concurrency

Use `web.serve(routes, options)` as the service boundary:

web.serve(routes, {
  host: '127.0.0.1',
  port: 3072,
  workers: 4,
  request_timeout_ms: 15000,
  max_body_bytes: 1048576
})

`web.service(...)` remains a pure descriptor for planning and documentation. `web.serve(...)` crosses the network boundary. By default it uses a poll-backed worker pool sized from the machine's online CPU count, so I/O can progress on multiple cores. For deterministic specs, pass an existing `listener` from `net.listen_loopback()` and `max_requests: 1` (or another finite number) so the server returns after serving the requested traffic.

The intended concurrency model is:

  • each listener worker handles lightweight request parsing and dispatch;
  • request handlers remain synchronous and fast;
  • slow work is explicitly pushed into `stdlib:jobs` or actors;
  • actors own shared in-memory state such as chat rooms, counters, and presence;
  • Jobs own CPU-heavy or slow background work.

`stdlib:web` does not secretly turn every request into a Job. That would make latency and ordering harder to reason about. Instead, the controller chooses when to enqueue work.

EHY form and shared-room example

This example shows the pieces you use for a small interactive app: EHY views, a form, dynamic time, and a shared room API that can later be backed by an actor. The dispatch version is socket-free so it is easy to test. The same routes are the ones you pass to `web.serve(...)`.

import 'stdlib:web'
import 'stdlib:ehy'

module Views
  fn layout(title, body)
    return ehy.render_file_with('examples/templates/layout.ehy', {title: title, body: body})
  end

  fn form()
    let form = """<form action="/hello" method="GET">
  <input name="name">
  <button>Say hello</button>
</form>"""
    return Views.layout('Hello', form)
  end

  fn hello(name, current)
    let template = """<h1>Hello <%= name %></h1>
<p>Now: <%= current %></p>"""
    return Views.layout('Hello ' + name, ehy.result_with(template, {name: name, current: current}))
  end
end

module Room
  fn snapshot(messages)
    return {count: messages.length, messages: messages}
  end

  fn add(messages, from, text)
    return [...messages, {from: from, text: text}]
  end
end

module PeopleController
  fn form(request)
    return web.html(Views.form())
  end

  fn hello(request)
    let name = web.query_param(request, 'name', 'friend')
    return web.html(Views.hello(name, now_ns()))
  end
end

module RoomController
  fn show(request)
    let snapshot = Room.snapshot([])
    return web.json(snapshot)
  end
end

module AppMiddleware
  fn add_app_context(request)
    return {...request, app_name: 'shared_room'}
  end
end

program
  let routes = web.routes([
    web.get('/', PeopleController.form),
    web.get('/hello', PeopleController.hello),
    web.get('/room', RoomController.show)
  ])

  let app = web.stack([
    web.request_id(),
    web.secure_headers(),
    web.middleware('app_context', AppMiddleware.add_app_context)
  ], routes)

  web.serve(app, {host: '127.0.0.1', port: 3072, workers: 4})
end

`examples/templates/layout.ehy`:

<!doctype html>
<html>
  <head><title><%= title %></title></head>
  <body><%= body %></body>
</html>

For a real multi-tab chat, make `Room` an actor with one mailbox-owned message list. Each browser request sends a message to that actor: direct messages include a target user, and broadcast messages append to the shared room. The controller should only validate, send to the actor, and render the current snapshot.

Live server smoke

Run a small app with `web.serve` and hit it with curl:

import 'stdlib:web'

module Health
  fn show(request)
    return web.json({ok: true})
  end
end

program
  let routes = web.routes([web.get('/health', Health.show)])
  web.serve(routes, {host: '127.0.0.1', port: 3072})
end

curl http://127.0.0.1:3072/health

For integration specs, use `max_requests` so the server exits:

let listener = net.listen_loopback().value
web.serve(routes, {listener: listener, max_requests: 1, workers: 2})

`workers` controls the accept/request worker pool. When omitted, the runtime uses the online CPU count (capped internally) rather than pinning all I/O to a single worker.
